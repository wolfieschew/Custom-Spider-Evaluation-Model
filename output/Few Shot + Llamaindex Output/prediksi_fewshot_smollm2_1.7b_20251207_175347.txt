SELECT T1.name FROM singer AS T1 JOIN song AS T2 ON T1.singer_id = T2.singer_id GROUP BY T1.name HAVING COUNT(*) > 1	singer
SELECT T1.Name, SUM(T2.Sales) AS Total_Sales FROM singer AS T1 JOIN song AS T2 ON T1.Singer_ID = T2.Singer_ID GROUP BY T1.Name ORDER BY Total_Sales DESC	singer
SELECT SUM(Sales) FROM song WHERE Singer_ID IN (SELECT Singer_ID FROM singer GROUP BY Singer_ID HAVING COUNT(*) > 1)	singer
